# ===========================================
# ⚙️ AGENT ARCHITECTURE
# ===========================================

[AGENTS]

# Planner Agent
[AGENTS.PLANNER]
role="Break down high-level Dreams into structured subtasks"
input="Dream markdown files from leader-brain/dreams/"
output="Step-by-step execution plan in the Brain state"
memory_access=true
tools=["leader-brain-fs"]

# Executor Agent
[AGENTS.EXECUTOR]
role="Executes tasks generated by planner using available tools"
input="Execution plan from Planner"
output="Completed actions, code changes, or API results"
memory_access=true
tools=["shell", "git", "vultr-inference"]

# Memory Agent
[AGENTS.MEMORY]
role="Manages the Leader Brain file system and markdown database"
input="New knowledge, results, mistakes, or patterns"
output="Updated markdown files in leader-brain/"
memory_access=true
tools=["mddb", "fs"]

# Critic Agent
[AGENTS.CRITIC]
role="Evaluates Executor outputs for accuracy and goal alignment"
input="Completed results from Executor"
output="Validation status or retry instructions"
memory_access=true
tools=["vultr-inference"]
